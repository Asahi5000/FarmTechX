#include <WiFi.h>
#include <HTTPClient.h>
#include <OneWire.h>
#include <DallasTemperature.h>

// === PIN CONFIG ===
#define ONE_WIRE_BUS 17   // DS18B20 data pin
#define EC_PIN 34         // Analog pin for EC sensor
#define PH_PIN 35         // Analog pin for pH module

// === WIFI CONFIG ===
const char* ssid = "Maejed_2.4g";          
const char* password = "Chijedjoydred20";  

// === SERVER CONFIG ===
const char* serverName = "http://192.168.1.3/FarmTechX/assets/php/api/save-sensors.php";

// === DS18B20 Setup ===
OneWire oneWire(ONE_WIRE_BUS);
DallasTemperature DS18B20(&oneWire);

void setup() {
  Serial.begin(115200);

  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nâœ… WiFi Connected!");
  Serial.print("ESP32 IP Address: ");
  Serial.println(WiFi.localIP());

  DS18B20.begin();
}

void loop() {
  // === Read Temperature ===
  DS18B20.requestTemperatures();
  float tempC = DS18B20.getTempCByIndex(0);

  // === Read EC Sensor ===
  int rawEC = analogRead(EC_PIN);
  float ecValue = (rawEC / 4095.0) * 5.0;  // Example scaling (calibrate later)

  // === Read pH Sensor ===
  int rawPH = analogRead(PH_PIN);
  float voltagePH = (rawPH / 4095.0) * 3.3;  
  // Typical conversion (needs calibration with buffer solutions)
  float phValue = 7 + ((2.5 - voltagePH) / 0.18);

  Serial.print("ğŸŒ¡ Temp: ");
  Serial.print(tempC);
  Serial.print(" Â°C | ğŸŒŠ EC: ");
  Serial.print(ecValue);
  Serial.print(" mS/cm | ğŸ§ª pH: ");
  Serial.println(phValue);

  if (WiFi.status() == WL_CONNECTED) {
    HTTPClient http;
    http.begin(serverName);
    http.addHeader("Content-Type", "application/x-www-form-urlencoded");

    String postData = "temperature=" + String(tempC, 2) + 
                      "&ec=" + String(ecValue, 2) + 
                      "&ph=" + String(phValue, 2);

    int httpResponseCode = http.POST(postData);

    if (httpResponseCode > 0) {
      Serial.print("âœ… HTTP Response: ");
      Serial.println(httpResponseCode);
      Serial.println("ğŸ“© " + http.getString());
    } else {
      Serial.print("âŒ Error: ");
      Serial.println(httpResponseCode);
    }
    http.end();
  }

  delay(1000);
}
