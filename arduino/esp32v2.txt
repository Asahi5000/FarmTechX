#include <WiFi.h>
#include <HTTPClient.h>
#include <OneWire.h>
#include <DallasTemperature.h>

#define SENSOR_PIN  17  // DS18B20 data pin

// ‚úÖ Your WiFi credentials
const char* ssid = "Maejed_2.4g";
const char* password = "Chijedjoydred20";

// ‚úÖ Your PC (XAMPP server) LAN IP - keep this updated
const char* serverName = "http://192.168.1.3/FarmTechX/assets/php/api/save-temp.php";

// ‚úÖ (Optional) Static IP for ESP32
IPAddress local_IP(192, 168, 1, 50);   // ESP32 fixed IP
IPAddress gateway(192, 168, 1, 1);     // Router IP
IPAddress subnet(255, 255, 255, 0);

OneWire oneWire(SENSOR_PIN);
DallasTemperature DS18B20(&oneWire);

void connectWiFi() {
  WiFi.disconnect(true);
  delay(1000);

  Serial.print("Connecting to WiFi: ");
  Serial.println(ssid);

  WiFi.begin(ssid, password);

  unsigned long startAttemptTime = millis();

  // Keep retrying for 15s
  while (WiFi.status() != WL_CONNECTED && millis() - startAttemptTime < 15000) {
    Serial.print(".");
    delay(500);
  }

  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("\n‚úÖ WiFi connected!");
    Serial.print("ESP32 IP Address: ");
    Serial.println(WiFi.localIP());
  } else {
    Serial.println("\n‚ùå WiFi connection failed. Restarting...");
    ESP.restart();
  }
}

void setup() {
  Serial.begin(115200);

  // ‚úÖ Apply static IP before connecting
  if (!WiFi.config(local_IP, gateway, subnet)) {
    Serial.println(‚ö†Ô∏è Static IP Failed to configure.");
  }

  connectWiFi();
  DS18B20.begin();
}

void loop() {
  if (WiFi.status() != WL_CONNECTED) {
    Serial.println("‚ö†Ô∏è WiFi lost. Reconnecting...");
    connectWiFi();
  }

  DS18B20.requestTemperatures();
  float tempC = DS18B20.getTempCByIndex(0);

  if (isnan(tempC)) {
    Serial.println("‚ö†Ô∏è Failed to read temperature!");
    return;
  }

  if (WiFi.status() == WL_CONNECTED) {
    HTTPClient http;
    http.begin(serverName);
    http.addHeader("Content-Type", "application/x-www-form-urlencoded");

    String postData = "temperature=" + String(tempC, 2);

    Serial.println("üå° Sending POST: " + postData);

    int httpResponseCode = http.POST(postData);

    if (httpResponseCode > 0) {
      Serial.print("‚úÖ HTTP Response code: ");
      Serial.println(httpResponseCode);

      String payload = http.getString();
      Serial.println("üì© Server Response: " + payload);
    } else {
      Serial.print("‚ùå Error code: ");
      Serial.println(httpResponseCode);
    }

    http.end();
  }

  delay(5000); // every 5 seconds
}
