#include <WiFi.h>
#include <HTTPClient.h>
#include <OneWire.h>
#include <DallasTemperature.h>

// === PIN CONFIG ===
#define ONE_WIRE_BUS 17   // DS18B20 data pin
#define EC_PIN 34         // Analog pin for EC sensor
#define PH_PIN 35         // Analog pin for pH sensor

// === WIFI CONFIG ===
const char* ssid = "Maejed_2.4g";          // WiFi SSID
const char* password = "Chijedjoydred20";  // WiFi Password

// === SERVER CONFIG ===
const char* serverName = "http://192.168.1.3/FarmTechX/assets/php/api/save-temp.php";

// === DS18B20 Setup ===
OneWire oneWire(ONE_WIRE_BUS);
DallasTemperature DS18B20(&oneWire);

void setup() {
  Serial.begin(115200);

  // Connect to WiFi
  Serial.print("Connecting to WiFi: ");
  Serial.println(ssid);
  WiFi.begin(ssid, password);

  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nâœ… WiFi Connected!");
  Serial.print("ESP32 IP Address: ");
  Serial.println(WiFi.localIP());

  DS18B20.begin();
}

void loop() {
  // === Read Temperature ===
  DS18B20.requestTemperatures();
  float tempC = DS18B20.getTempCByIndex(0);

  // === Read EC Sensor ===
  int rawEC = analogRead(EC_PIN);
  float voltageEC = (rawEC / 4095.0) * 3.3; // ESP32 ADC (0â€“4095) mapped to 0â€“3.3V
  float ecValue = voltageEC * 10.0; // âš ï¸ Example scaling â†’ replace with calibration

  // === Read pH Sensor ===
  int rawPH = analogRead(PH_PIN);
  float voltagePH = (rawPH / 4095.0) * 3.3; // Convert ADC to voltage
  float phValue = 7 + ((2.5 - voltagePH) / 0.18); // âš ï¸ Example formula â†’ calibrate properly

  // === Debugging Serial Output ===
  Serial.print("ğŸŒ¡ Temp: ");
  Serial.print(tempC);
  Serial.print(" Â°C | ğŸŒŠ EC: ");
  Serial.print(ecValue);
  Serial.print(" mS/cm | ğŸ§ª pH: ");
  Serial.println(phValue);

  // === Send to Server ===
  if (WiFi.status() == WL_CONNECTED) {
    HTTPClient http;
    http.begin(serverName);
    http.addHeader("Content-Type", "application/x-www-form-urlencoded");

    // Send temperature + EC + pH
    String postData = "temperature=" + String(tempC, 2) + "&ec=" + String(ecValue, 2) + "&ph=" + String(phValue, 2);

    Serial.println("ğŸ“¤ Sending POST: " + postData);
    int httpResponseCode = http.POST(postData);

    if (httpResponseCode > 0) {
      Serial.print("âœ… HTTP Response code: ");
      Serial.println(httpResponseCode);
      String payload = http.getString();
      Serial.println("ğŸ“© Server Response: " + payload);
    } else {
      Serial.print("âŒ Error code: ");
      Serial.println(httpResponseCode);
    }
    http.end();
  } else {
    Serial.println("âš ï¸ WiFi Disconnected. Reconnecting...");
    WiFi.begin(ssid, password);
  }

  delay(2000); // send every 2 seconds
}
