#include <WiFi.h>
#include <HTTPClient.h>
#include <OneWire.h>
#include <DallasTemperature.h>
#include "DFRobot_ESP_EC.h"
#include <EEPROM.h>

// ========== Pin Config ==========
#define TEMP_PIN 17   // DS18B20 data pin
#define EC_PIN   34   // EC analog pin

// ========== WiFi ==========
const char* ssid = "Maejed_2.4g";
const char* password = "Chijedjoydred20";

// Server (XAMPP / PHP)
const char* serverName = "http://192.168.1.3/FarmTechX/assets/php/api/save-temp.php";

// Static IP
IPAddress local_IP(192, 168, 1, 50);
IPAddress gateway(192, 168, 1, 1);
IPAddress subnet(255, 255, 255, 0);

// ========== Sensors ==========
OneWire oneWire(TEMP_PIN);
DallasTemperature DS18B20(&oneWire);
DFRobot_ESP_EC ec;

// ========== WiFi Connect ==========
void connectWiFi() {
  WiFi.disconnect(true);
  delay(1000);

  Serial.print("Connecting to WiFi: ");
  Serial.println(ssid);

  WiFi.begin(ssid, password);
  unsigned long startAttemptTime = millis();

  // Try for 15s
  while (WiFi.status() != WL_CONNECTED && millis() - startAttemptTime < 15000) {
    Serial.print(".");
    delay(500);
  }

  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("\n‚úÖ WiFi connected!");
    Serial.print("ESP32 IP Address: ");
    Serial.println(WiFi.localIP());
  } else {
    Serial.println("\n‚ùå WiFi connection failed. Retrying in 5s...");
    delay(5000);
    connectWiFi(); // keep retrying
  }
}

void setup() {
  Serial.begin(115200);

  // Apply static IP
  if (!WiFi.config(local_IP, gateway, subnet)) {
    Serial.println("‚ö†Ô∏è Static IP Failed to configure.");
  }

  connectWiFi();

  // Start sensors
  DS18B20.begin();
  ec.begin();
}

void loop() {
  // Ensure WiFi stays connected
  if (WiFi.status() != WL_CONNECTED) {
    Serial.println("‚ö†Ô∏è WiFi lost. Reconnecting...");
    connectWiFi();
  }

  // ‚úÖ Temperature
  DS18B20.requestTemperatures();
  float tempC = DS18B20.getTempCByIndex(0);

  // ‚úÖ EC sensor
  int ecRaw = analogRead(EC_PIN);
  float voltage = ecRaw * (3.3 / 4095.0); // ADC ‚Üí voltage
  float ecValue = ec.readEC(voltage, tempC);

  Serial.print("üå° Temperature: ");
  Serial.print(tempC);
  Serial.println(" ¬∞C");

  Serial.print("üíß EC: ");
  Serial.print(ecValue, 3);
  Serial.println(" ms/cm");

  // ‚úÖ Send to server
  if (WiFi.status() == WL_CONNECTED) {
    HTTPClient http;
    http.begin(serverName);
    http.addHeader("Content-Type", "application/x-www-form-urlencoded");

    String postData = "temperature=" + String(tempC, 2) + "&ec=" + String(ecValue, 3);
    Serial.println("üì§ Sending POST: " + postData);

    int httpResponseCode = http.POST(postData);
    if (httpResponseCode > 0) {
      Serial.print("‚úÖ HTTP Response code: ");
      Serial.println(httpResponseCode);
      String payload = http.getString();
      Serial.println("üì© Server Response: " + payload);
    } else {
      Serial.print("‚ùå Error code: ");
      Serial.println(httpResponseCode);
    }

    http.end();
  }

  delay(2000); // every 2s
}
